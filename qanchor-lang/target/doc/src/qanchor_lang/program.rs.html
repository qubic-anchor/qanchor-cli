6/// Trait for QAnchor programs

19/// Program instruction dispatcher
20/// 
21/// This is used internally by the `#[program]` macro to dispatch
22/// instructions to their handlers.
23pub struct InstructionDispatcher;
24
25impl InstructionDispatcher {
26    /// Dispatch an instruction to its handler
27    pub fn dispatch<'info, T>(
28        program_id: &'info [u8; 32],
29        accounts: &'info [AccountInfo<'info>],
30        _instruction_data: &[u8],
31        handler: fn(Context<'info, T>) -> Result<(), ErrorCode>,
32    ) -> Result<(), ErrorCode>
33    where
34        T: crate::accounts::Accounts<'info>,
35    {
36        // Parse instruction data to extract parameters
37        let (accounts_data, remaining_accounts) = Self::parse_accounts(accounts)?;
38        
39        // Validate and deserialize accounts
40        let mut account_iter = accounts_data.iter().peekable();
41        let validated_accounts = T::try_accounts(
42            program_id,
43            &mut account_iter,
44            remaining_accounts,
45        )?;
46
47        // Create context
48        let ctx = Context::new(validated_accounts, program_id, remaining_accounts);
49
50        // Execute the instruction handler
51        handler(ctx)
52    }
53
54    /// Parse accounts into validated accounts and remaining accounts
55    fn parse_accounts<'info>(accounts: &'info [AccountInfo<'info>]) -> Result<(&'info [AccountInfo<'info>], &'info [AccountInfo<'info>]), ErrorCode> {
56        // For now, return all accounts as validated accounts
57        // In a real implementation, this would parse the account metadata
58        // to determine which accounts are for validation and which are remaining
59        Ok((accounts, &[]))
60    }
61}

79/// Instruction parameter extraction
80pub struct InstructionParams;
81
82impl InstructionParams {
83    /// Extract parameters from instruction data
84    pub fn extract<T>(instruction_data: &[u8]) -> Result<T, ErrorCode>
85    where
86        T: serde::de::DeserializeOwned,
87    {
88        serde_json::from_slice(instruction_data)
89            .map_err(|_| ErrorCode::InvalidInstructionData)
90    }
91}
